Create a Server with Gulp

First install Gulp with:

npm install --save-dev gulp gulp-connect gulp-watch

Then add the gulp tasks to the gulpfile.js in the root path of the project:

var gulp    = require('gulp');
var watch   = require('gulp-watch')
var connect = require('gulp-connect');

gulp.task('webserver', function() {
  connect.server({
    livereload: true,
    root: 'src'
  });
});

gulp.task('default', ['webserver', 'watch']);

To reload the server if a file have been update, for example all .html files. Add a task that reload the server and an other that watch for changes:

gulp.task('html', function() {
gulp.src('./**/*.html')
  .pipe(connect.reload());
});

gulp.task('watch', function() {
gulp.watch(['./**/*.html'], ['html']);
});

Build a Distribution

To compile and uglify the javascript gulp-uglify can be used:

// compress and copy the javascript files to the project
gulp.task('compress', function() {
  gulp.src('src/js/*.js')
    .pipe(uglify())
    .pipe(gulp.dest('dist/js/'));
});

To copy things to the distribution gulp-copy can be used:

// copy the html files to the project
gulp.task('copy-html', function() {
  gulp.src([
    'src/index.html',
    'src/templates/*.html',
    'src/modules/menu/templates/*.html'
  ]).pipe(copy('dist', {prefix: 1}));
})

To replace text in a file for exaple change the angular source to a cns, gulp-replace can be used:

// replace the linkto the angular project to an cns server
gulp.task('angular-src-link', function() {
  gulp
    .src('src/index.html')
    .pipe(replace(
      "libs/angular.js",
      "https://ajax.googleapis.com/ajax/libs/angularjs/1.5.6/angular.min.js"))
    .pipe(gulp.dest('dist'));
});
